create table public.recipe_lines (
  id uuid not null default gen_random_uuid (),
  parent_item_id uuid not null,
  line_type text not null,
  child_item_id uuid null,
  quantity numeric null,
  unit text null,
  labor_role text null,
  minutes numeric null,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  specific_child text null,
  last_change text null,
  user_id uuid null,
  tenant_id uuid not null,
  constraint recipe_lines_pkey primary key (id),
  constraint recipe_lines_child_item_id_fkey foreign KEY (child_item_id) references items (id),
  constraint recipe_lines_user_id_fkey foreign KEY (user_id) references users (id) on delete CASCADE,
  constraint recipe_lines_parent_item_id_fkey foreign KEY (parent_item_id) references items (id) on delete CASCADE,
  constraint recipe_lines_tenant_id_fkey foreign KEY (tenant_id) references tenants (id) on delete CASCADE,
  constraint chk_recipe_lines_labor check (
    (
      (line_type <> 'labor'::text)
      or (
        (minutes is not null)
        and (minutes > (0)::numeric)
        and (child_item_id is null)
        and (quantity is null)
        and (unit is null)
      )
    )
  ),
  constraint recipe_lines_line_type_check check (
    (
      line_type = any (array['ingredient'::text, 'labor'::text])
    )
  ),
  constraint chk_recipe_lines_ingredient check (
    (
      (line_type <> 'ingredient'::text)
      or (
        (child_item_id is not null)
        and (quantity is not null)
        and (quantity > (0)::numeric)
        and (unit is not null)
        and (minutes is null)
        and (labor_role is null)
      )
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_recipe_lines_user_id on public.recipe_lines using btree (user_id) TABLESPACE pg_default;

create index IF not exists idx_recipe_lines_parent on public.recipe_lines using btree (parent_item_id) TABLESPACE pg_default;

create index IF not exists idx_recipe_lines_tenant_id on public.recipe_lines using btree (tenant_id) TABLESPACE pg_default;

create trigger trigger_update_recipe_lines_updated_at BEFORE
update on recipe_lines for EACH row
execute FUNCTION update_recipe_lines_updated_at ();


create table public.items (
  id uuid not null default gen_random_uuid (),
  name text null,
  item_kind text not null,
  is_menu_item boolean not null default false,
  proceed_yield_amount numeric null,
  proceed_yield_unit text null,
  notes text null,
  each_grams numeric null,
  base_item_id uuid null,
  deprecated timestamp with time zone null,
  deprecation_reason text null,
  wholesale numeric null,
  retail numeric null,
  user_id uuid null,
  tenant_id uuid not null,
  responsible_user_id uuid null,
  constraint items_pkey primary key (id),
  constraint items_user_id_fkey foreign KEY (user_id) references users (id) on delete CASCADE,
  constraint items_base_item_id_fkey foreign KEY (base_item_id) references base_items (id) on delete set null,
  constraint items_responsible_user_id_fkey foreign KEY (responsible_user_id) references users (id) on delete set null,
  constraint items_tenant_id_fkey foreign KEY (tenant_id) references tenants (id) on delete CASCADE,
  constraint chk_items_yield_unit_mass check (
    (
      (proceed_yield_unit is null)
      or (
        proceed_yield_unit = any (
          array[
            'g'::text,
            'kg'::text,
            'lb'::text,
            'oz'::text,
            'each'::text
          ]
        )
      )
    )
  ),
  constraint chk_items_menu_must_be_prepped check (
    (
      (not is_menu_item)
      or (item_kind = 'prepped'::text)
    )
  ),
  constraint items_each_grams_check check ((each_grams > (0)::numeric)),
  constraint items_item_kind_check check (
    (
      item_kind = any (array['raw'::text, 'prepped'::text])
    )
  ),
  constraint chk_items_prepped_fields_new check (
    (
      (item_kind <> 'prepped'::text)
      or (
        (proceed_yield_amount is not null)
        and (proceed_yield_amount > (0)::numeric)
        and (proceed_yield_unit is not null)
        and (base_item_id is null)
      )
    )
  ),
  constraint chk_items_prepped_has_name check (
    (
      (item_kind <> 'prepped'::text)
      or (name is not null)
    )
  ),
  constraint chk_items_raw_fields_new check (
    (
      (item_kind <> 'raw'::text)
      or (
        (base_item_id is not null)
        and (proceed_yield_amount is null)
        and (proceed_yield_unit is null)
      )
    )
  ),
  constraint chk_items_raw_has_base_item check (
    (
      (item_kind <> 'raw'::text)
      or (base_item_id is not null)
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_items_deprecated on public.items using btree (deprecated) TABLESPACE pg_default;

create index IF not exists idx_items_item_kind on public.items using btree (item_kind) TABLESPACE pg_default;

create index IF not exists idx_items_is_menu_item on public.items using btree (is_menu_item) TABLESPACE pg_default;

create index IF not exists idx_items_user_id on public.items using btree (user_id) TABLESPACE pg_default;

create index IF not exists idx_items_base_item on public.items using btree (base_item_id) TABLESPACE pg_default;

create index IF not exists idx_items_tenant_id on public.items using btree (tenant_id) TABLESPACE pg_default;

create index IF not exists idx_items_responsible_user_id on public.items using btree (responsible_user_id) TABLESPACE pg_default;

create unique INDEX IF not exists idx_items_name_tenant_id_unique_active on public.items using btree (name, tenant_id) TABLESPACE pg_default
where
  (deprecated is null);
